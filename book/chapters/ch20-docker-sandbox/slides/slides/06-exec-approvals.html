// exec-approvals.ts 核心逻辑
class ExecApprovalStore {
  private approvals = new Map<string, Set>()
  isApproved(sessionId, cmdPattern) {
    return this.approvals.get(sessionId)?.has(cmdPattern)
  }
  approve(sessionId, cmdPattern) {
    // 记忆：pattern 而非完整命令
    this.approvals.get(sessionId).add(cmdPattern)
  }
  clearSession(sessionId) {
    this.approvals.delete(sessionId) // session 结束重置
  }
}
