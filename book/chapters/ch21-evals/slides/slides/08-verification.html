# self_verify.py — 借鉴 CC nudge 机制

SELF_VERIFY_PROMPT = """
你刚完成了：{task_description}
你的输出：{output}

评估（0-10 分）：
- completeness 完整性
- accuracy     准确性
- format       格式

< 8 分列出具体缺少什么。
"""

class SelfVerifier:
    def verify(self, task: str, output: str):
        result = haiku.complete(SELF_VERIFY_PROMPT)

        if result.completeness < 8:
            raise IncompleteTaskError(
                result.missing_items
            )
        return result
