# src/observability/tracing.py from opentelemetry import trace from opentelemetry.sdk.trace import TracerProvider from opentelemetry.exporter.otlp.proto.grpc... import OTLPSpanExporter def setup_tracing(service_name="lena", otlp_endpoint="http://localhost:4317"): resource = Resource.create({ "service.name": service_name, "service.version": "1.8.0", }) provider = TracerProvider(resource=resource) exporter = OTLPSpanExporter( endpoint=otlp_endpoint, insecure=True ) provider.add_span_processor( BatchSpanProcessor(exporter)) trace.set_tracer_provider(provider) return trace.get_tracer(service_name) # 使用：with tracer.start_as_current_span("llm_call") as span: # span.set_attribute("model", "claude-sonnet-4-6")
