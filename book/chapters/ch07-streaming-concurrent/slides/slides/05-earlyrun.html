class StreamingToolExecutor {
  private pending: Map<string, Promise> = new Map()

  // query.ts:842 — tool_use 块流式到达时调用
  addTool(block: ToolUseBlock): void {
    if (this.pending.size >= MAX_CONCURRENCY) {
      this.queue.push(block)  // 超上限排队
      return
    }
    // 立即启动，不等流结束！
    const promise = this.executeTool(block)
    this.pending.set(block.id, promise)
  }

  // 流结束后调用，等待全部工具完成
  async getRemainingResults(): AsyncGenerator {
    for (const [id, promise] of this.pending) {
      const result = await promise
      yield buildToolResult(result)
    }
  }
}
